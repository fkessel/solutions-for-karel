void harvestTheField(){
    repeat (4){
        pickUpBeeper();
    }
    iHaveNoBetterSolutionAtTheMoment();
    repeat (9){
        pickUpBeeper();
    }
    iHaveNoBetterSolutionAtTheMoment();
    repeat (4){
        pickUpBeeper();
    }
}

void pickUpBeeper(){
    
    if (beeperAhead()){
        moveForward();
        pickBeeper();
        turnRight();
        moveForward();
        turnLeft();
    }
    else if (leftIsClear() && frontIsClear() && rightIsClear()){
        turnAround();
    }
}

void iHaveNoBetterSolutionAtTheMoment(){
    repeat (2){
        moveForward();
        moveForward();
        turnLeft();
    }
}
