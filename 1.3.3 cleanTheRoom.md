void cleanTheRoom(){
    repeat (4){
        forward();
        rightSide();
        forward();
        leftSide();
    }
    forward();
    rightSide();
    forward();
}

void forward(){
    if (onBeeper()){
        pickBeeper();
    }
    repeat (9){
        moveForward();
        if (onBeeper()){
            pickBeeper();
        }
    }
}

void rightSide(){
    turnLeft();
    moveForward();
    turnLeft();
}

void leftSide(){
    turnRight();
    moveForward();
    turnRight();   
}
